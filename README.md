# Maior-Valor
Exercício de logica de programação 2

Maior valor
Gabarito do Exercício 2
programa

{

   funcao inicio(){

      real valor1, valor2

      escreva ("Informe a nota da valor 1: ")

      leia(valor1)

      escreva ("Informe a nota da valor 2: ")

      leia(valor2)

      se(valor1 > valor2) {

         escreva ("O valor 1 é o maior")

      }

      senao{

         escreva ("O valor 2 é o maior")

      }

   }

}
